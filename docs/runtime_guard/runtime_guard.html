runtime_guard.runtime_guard

Runtime Guard

This module provides utilities to monitor hardware usage and prevent server stalls during long-running loops, training jobs, or other iterative processes.

Features

  • RAM usage monitoring
  • CPU usage monitoring
  • GPU memory monitoring (PyTorch / TensorFlow)
  • Mean loop time tracking
  • Watchdog timer
  • Python memory leak detection

Usage Patterns

Standard:

import runtime_guard as run

guard = run.RuntimeGuard()

for epoch in epochs:
    for idx, x, y in data:
        guard.start_loop()

        # do work

        guard.update()

    if validate:
        guard.pause()
        # validation

Iterator-Method:

import runtime_guard as run

guard = run.RuntimeGuard()

for epoch in epochs:
    for idx, x, y in guard(data):
        # do work

    with run.Pause(guard):
        if validate:
            # or just: guard.pause()
            # validation

Context-Method:

import runtime_guard as run

guard = run.RuntimeGuard()

for epoch in epochs:
    for idx, x, y in data:
        with run.Run(guard):
            # do work

    with run.Pause(guard):
        if validate:
            # or just: guard.pause()
            # validation

356    def update(self):
357        """
358        ## Update RuntimeGuard
359
360        Executes all enabled checks:
361        - RAM
362        - CPU
363        - GPU
364        - Mean loop time
365        - Watchdog
366        - Memory leak detection
367
368        Handles:
369        - Printing
370        - Logging
371        - Abort conditions
372        """
373        if self.cur_call_n < self.warm_up_iter or \
374            self.cur_call_n % self.update_every_x_calls != 0:
375            self.last_heartbeat = time.time()
376            self.cur_call_n += 1
377            return
378
379        # make checks
380        if self.make_ram_check:
381            ram_result = self.check_ram()
382        else:
383            ram_result = "    - No RAM Checking"
384
385        if self.make_cpu_check:
386            cpu_result = self.check_cpu()
387        else:
388            cpu_result = "    - No CPU Checking"
389
390        if self.make_gpu_check:
391            gpu_result = self.check_gpu()
392        else:
393            gpu_result = "    - No GPU Checking"
394
395        if self.make_mean_loop_time_check:
396            self.next_loop()
397            mean_loop_time_result = self.check_mean_loop_time()
398        else:
399            mean_loop_time_result = "    - No Mean Loop Time Checking"
400
401        if self.abort_event.is_set():
402            self.exit(msg="[ABORT] Watchdog timeout")
403        else:
404            watchdog_result = "    - Watchdog is still fine!"
405        
406        if self.make_leak_check:
407            leak_result = self.check_leak()
408        else:
409            leak_result = "    - No Leak Checking"
410
411        # printing and logging
412        self.make_print_entry(ram_result=ram_result, cpu_result=cpu_result, 
413                              gpu_result=gpu_result, mean_loop_time_result=mean_loop_time_result, 
414                              watchdog_result=watchdog_result, leak_result=leak_result)

416                            gpu_result=gpu_result, mean_loop_time_result=mean_loop_time_result, 
417                            watchdog_result=watchdog_result, leak_result=leak_result)
418        
419        # update vars
420        self.cur_call_n += 1
421        self.last_heartbeat = time.time()

484    def check_gpu(self):
485        """
486        ## Resource Check
487
488        Performs the respective resource check and aborts if thresholds are exceeded.
489
490        ### Returns
491        - `str`: Human-readable status message
492        """
493        if _HAS_TORCH and torch.cuda.is_available():
494            used = torch.cuda.memory_allocated()
495            total = torch.cuda.get_device_properties(0).total_memory
496            usage_ratio = used / total
497            if usage_ratio > self.max_gpu_usage:
498                abort_txt = f"[ABORT] GPU memory full (PyTorch)"
499                self.make_abort_log(abort_message=abort_txt)
500                print(abort_txt)
501                self.exit(msg=abort_txt)
502            return f"    - GPU checked (PyTorch)\n        -> usage = {used/(1024**3):.2f} GB / {total/(1024**3):.2f} GB ({usage_ratio*100:.2f}%)"
503
504        elif _HAS_TF:
505            gpus = tf.config.experimental.list_physical_devices('GPU')
506            if gpus:
507                try:
508                    details = tf.config.experimental.get_memory_info('GPU:0')
509                    used = details['current']
510                    total = details['peak']
511                    usage_ratio = used / total
512                    if usage_ratio > self.max_gpu_usage:
513                        abort_txt = f"[ABORT] GPU memory full (TensorFlow)"
514                        self.make_abort_log(abort_message=abort_txt)
515                        print(abort_txt)
516                        self.exit(msg=abort_txt)
517                    return f"    - GPU checked (TensorFlow)\n        -> usage = {used/(1024**3):.2f} GB / {total/(1024**3):.2f} GB ({usage_ratio*100:.2f}%)"
518                except Exception as e:
519                    return f"    - GPU check (TensorFlow) failed: {e}"
520            else:
521                return "    - GPU checked (TensorFlow) -> no GPU found"
522
523        else:
524            return "    - GPU check skipped (no supported framework installed)"

527    def check_mean_loop_time(self):
528        """
529        ## Mean Loop Time Check
530
531        Compares current mean loop duration against the baseline.
532
533        Aborts if slowdown exceeds the configured factor.
534
535        ### Returns
536        - `str`: Status report
537        """
538        if not self.goal_mean_loop_time_collected:
539            return "    - Mean Loop Time not yet collected"
540        
541        cur_mean_time = self.get_mean_loop_time()
542        if cur_mean_time <= 0:
543            return "    - Mean Loop Time not yet collected"
544        
545        duration_factor_percentage = cur_mean_time / self.goal_mean_loop_time
546        if duration_factor_percentage > self.max_duration_factor_percentage:
547            abort_txt = f"[ABORT] Process got slowed down by {duration_factor_percentage} compared to the beginning"
548            self.make_abort_log(abort_message=abort_txt)
549            print(abort_txt)

643        """
644        ## Memory Leak Detection
645
646        Uses `tracemalloc` snapshots to detect memory growth.
647
648        Aborts if:
649        - Absolute growth exceeds limit
650        - Relative growth exceeds ratio
651
652        ### Returns
653        - `str`: Leak summary
654        """
655        snapshot = tracemalloc.take_snapshot()
656
657        if self.last_snapshot is None:
658            self.last_snapshot = snapshot
659            return "    - Leak not checked because there is no last snapshot"
660
661        stats = snapshot.compare_to(self.last_snapshot, 'lineno')
662
663        # current total growth of only increasing memory
664        total_growth_bytes = sum(
665            stat.size_diff for stat in stats if stat.size_diff > 0
666        )
667
668        total_growth_mb = total_growth_bytes / (1024 ** 2)
669
670        # previous total growth of only increasing memory
671        prev_total_bytes = sum( stat.size for stat in self.last_snapshot.statistics('lineno') )
672
673        # calc ratio
674        if prev_total_bytes > 0:
675            growth_ratio = total_growth_bytes / prev_total_bytes
676        else:
677            growth_ratio = 0.0
678
679        txt = f"\n        -> Growth = {total_growth_mb:.2f} MB"
680        txt += f"\n        -> Ratio  = {growth_ratio * 100:.2f}%"
681        txt += f"\n        -> Top leak sources:"
682        for stat in stats[:10]:
683            if stat.size_diff > 0:
684                txt += f"\n            {stat}"
685
686        if total_growth_mb > self.max_leak_mb or \
687           growth_ratio > self.max_leak_ratio:
688            self.last_snapshot = snapshot
689            raise RuntimeError(
690                f"[ABORT] Python memory leak detected. Details:\n{txt}"
691            )
692
693        # No leak -> update snapshot + return txt
694        self.last_snapshot = snapshot
695
696        return "    - Leak checked" + txt

528    def check_mean_loop_time(self):
529        """
530        ## Mean Loop Time Check
531
532        Compares current mean loop duration against the baseline.
533
534        Aborts if slowdown exceeds the configured factor.
535
536        ### Returns
537        - `str`: Status report
538        """
539        if not self.goal_mean_loop_time_collected:
540            return "    - Mean Loop Time not yet collected"
541        
542        cur_mean_time = self.get_mean_loop_time()
543        if cur_mean_time <= 0:
544            return "    - Mean Loop Time not yet collected"
545        
546        duration_factor_percentage = cur_mean_time / self.goal_mean_loop_time
547        if duration_factor_percentage > self.max_duration_factor_percentage:
548            abort_txt = f"[ABORT] Process got slowed down by {duration_factor_percentage} compared to the beginning"
549            self.make_abort_log(abort_message=abort_txt)
550            print(abort_txt)
551            self.exit(msg=abort_txt)
552
553        return f"    - Mean Loop Time checked\n        -> mean time trend = {duration_factor_percentage*100:.2f}% compared to the goal loop time\n        -> current mean loop duration {cur_mean_time:.2f} / base mean loop duration {self.goal_mean_loop_time:.2f}"

555    def next_loop(self):
556        """
557        Marks the end of the current loop iteration and updates loop time statistics.
558
559        Tracks duration of the loop, updates the mean loop time window, 
560        and sets the goal mean loop time once enough iterations have been collected.
561        """
562        if self.cur_loop_start_time is None:
563            # skip the first loop and start measuring
564            self.cur_loop_start_time = time.time()
565        else:
566            # get time duration
567            duration = time.time() - self.cur_loop_start_time
568
569            # add new mean value
570            self.mean_loop_time_window.append(duration)
571
572            # update goal mean loop time
573            if not self.goal_mean_loop_time_collected and len(self.mean_loop_time_window) >= self.mean_loop_time_window_size:
574                self.goal_mean_loop_time = self.get_mean_loop_time()
575                self.goal_mean_loop_time_collected = True
576
577            self.cur_loop_start_time = time.time()

599    def check_watchdog_timer(self):
600        """
601        ## Watchdog Monitor
602
603        Runs a background daemon thread that:
604        - Monitors loop heartbeats
605        - Detects stalled execution
606        - Triggers abort on timeout
607        """
608        def watchdog_func():
609            start_time = time.time()
610            while True:
611                time.sleep(self.watchdog_seconds_waittime)
612                if not self.watchdog_should_wait:
613                    duration = time.time() - self.last_heartbeat
614                    if duration > self.max_watchdog_seconds_timeout:
615                        abort_txt = f"[ABORT] Training stalled -> Watchdog got delayed by {duration} seconds"
616                        self.make_abort_log(abort_message=abort_txt)
617                        print(abort_txt)
618                        self.abort_event.set()
619                        raise RuntimeError(abort_txt)  # will only exit the thread
620
621        # runs alone (no reference to it needed) but get terminated when the pipeline is finish
622        self.watchdog = threading.Thread(target=watchdog_func, daemon=True)
623        self.watchdog.start()

643    def check_leak(self):
644        """
645        ## Memory Leak Detection
646
647        Uses `tracemalloc` snapshots to detect memory growth.
648
649        Aborts if:
650        - Absolute growth exceeds limit
651        - Relative growth exceeds ratio
652
653        ### Returns
654        - `str`: Leak summary
655        """
656        snapshot = tracemalloc.take_snapshot()
657
658        if self.last_snapshot is None:
659            self.last_snapshot = snapshot
660            return "    - Leak not checked because there is no last snapshot"
661
662        stats = snapshot.compare_to(self.last_snapshot, 'lineno')
663
664        # current total growth of only increasing memory
665        total_growth_bytes = sum(
666            stat.size_diff for stat in stats if stat.size_diff > 0
667        )
668
669        total_growth_mb = total_growth_bytes / (1024 ** 2)
670
671        # previous total growth of only increasing memory
672        prev_total_bytes = sum( stat.size for stat in self.last_snapshot.statistics('lineno') )
673
674        # calc ratio
675        if prev_total_bytes > 0:
676            growth_ratio = total_growth_bytes / prev_total_bytes
677        else:
678            growth_ratio = 0.0
679
680        txt = f"\n        -> Growth = {total_growth_mb:.2f} MB"
681        txt += f"\n        -> Ratio  = {growth_ratio * 100:.2f}%"
682        txt += f"\n        -> Top leak sources:"
683        for stat in stats[:10]:
684            if stat.size_diff > 0:
685                txt += f"\n            {stat}"
686
687        if total_growth_mb > self.max_leak_mb or \
688           growth_ratio > self.max_leak_ratio:
689            self.last_snapshot = snapshot
690            raise RuntimeError(
691                f"[ABORT] Python memory leak detected. Details:\n{txt}"
692            )
693
694        # No leak -> update snapshot + return txt
695        self.last_snapshot = snapshot
696
697        return "    - Leak checked" + txt

658        if self.last_snapshot is None:
659            self.last_snapshot = snapshot
660            return "    - Leak not checked because there is no last snapshot"
661
662        stats = snapshot.compare_to(self.last_snapshot, 'lineno')
663
664        # current total growth of only increasing memory
665        total_growth_bytes = sum(
666            stat.size_diff for stat in stats if stat.size_diff > 0
667        )
668
669        total_growth_mb = total_growth_bytes / (1024 ** 2)
670
671        # previous total growth of only increasing memory
672        prev_total_bytes = sum( stat.size for stat in self.last_snapshot.statistics('lineno') )
673
674        # calc ratio
675        if prev_total_bytes > 0:
676            growth_ratio = total_growth_bytes / prev_total_bytes
677        else:
678            growth_ratio = 0.0
679
680        txt = f"\n        -> Growth = {total_growth_mb:.2f} MB"
681        txt += f"\n        -> Ratio  = {growth_ratio * 100:.2f}%"
682        txt += f"\n        -> Top leak sources:"
683        for stat in stats[:10]:
684            if stat.size_diff > 0:
685                txt += f"\n            {stat}"
686
687        if total_growth_mb > self.max_leak_mb or \
688           growth_ratio > self.max_leak_ratio:
689            self.last_snapshot = snapshot
690            raise RuntimeError(
691                f"[ABORT] Python memory leak detected. Details:\n{txt}"
692            )
693
694        # No leak -> update snapshot + return txt
695        self.last_snapshot = snapshot
696
697        return "    - Leak checked" + txt

  • Absolute growth exceeds limit
  • Relative growth exceeds ratio
